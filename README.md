```bash
npm install --save card-memory-react
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'card-memory-react'
import 'card-memory-react/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
